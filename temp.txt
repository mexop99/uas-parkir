SELECT
    pe.*,
    pc.nama_kendaraan,
    usr.username
FROM
    parking_enter pe
INNER JOIN parking_cost pc ON
    pe.parking_cost_id = pc.id
INNER JOIN users usr ON
    usr.id = pe.user_id ORDER BY pe.id ASC
